5.2) Se ingresan notas numéricas de 0 a 10. El proceso de carga finaliza cuando se detecta un número fuera de rango (negativo o mayor que 10). La computadora muestra el promedio de las notas.

Resolución:

*
* @author 37adr
*/
import java.util.Scanner;
public class ejercicio52 {
public static void main(String[] args){

double numero=0;
int cont=0;
double promedio=0;
double suma=0;

Scanner teclado=new Scanner(System.in);


System.out.println("ingrese la nota");
numero=teclado.nextInt();

while(numero<""10&&numero>0){
cont++;
suma+=numero;
promedio=suma/cont;

System.out.println("ingrese otra nota");
numero=teclado.nextInt();
}
System.out.println("las notas que ingresaste son: "+cont);
System.out.println("el prmedio de las nota es: "+promedio);
}

}
