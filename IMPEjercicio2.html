Ejercicio 1):

2)Ingresar or teclado el nombre y apellido y tres notas del alumno mostrar en pantalla el nombre apellido y promedio

Resolución:

import java.util.Scanner;
public class Promedio { public static void main(String[] args){

String nombre="";
String apellido="";
int nota1=0;
int nota2=0;
int nota3=0;
double promedio=0;
Scanner teclado=new Scanner(System.in);

System.out.println("ingrese el nombre");

nombre=teclado.next();

System.out.println("ingrese el Apellido");

apellido=teclado.next();

System.out.println("ingrese la nota 1");

nota1=teclado.nextInt();

System.out.println("ingrese la nota 2");

nota2=teclado.nextInt();

System.out.println("ingrese la nota 3");

nota3=teclado.nextInt();

promedio=(nota1+nota2+nota3)/3;
System.out.println(nombre+" "+apellido+" su promedio es: "+promedio);

}
}
